import streamlit as st
from chatbot import ask_llm, analyze_sentiment

st.title("LLM Vehicle Maintenance Assistant")

query = st.text_area("Enter Vehicle Problem")

if st.button("Analyze Issue"):

    result = ask_llm(query)

    priority = analyze_sentiment(query)

    st.subheader("Issue Priority")
    st.warning(priority)

    st.subheader("AI Diagnosis Report")
    st.write(result)
